3. **Add get_current_client to backend/app/core/dependencies.py**:
   ```python
   async def get_current_client(
       request: Request,
       db: Annotated[AsyncSession, Depends(get_db)],
   ) -> "Client":
       token = request.cookies.get("client_session")
       if not token:
           raise HTTPException(401, "Not authenticated")
       result = await db.execute(
           select(ClientSession)
           .where(ClientSession.token == token, ClientSession.is_verified == True, ClientSession.expires_at > func.now())
           .options(selectinload(ClientSession.client))
       )
       session = result.scalar_one_or_none()
       if not session:
           raise HTTPException(401, "Session expired")
       return session.client
   ```
