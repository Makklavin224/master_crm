<!-- Backend API contracts from Plan 01 (master-authenticated endpoints) -->

Master endpoints (Bearer token in Authorization header):

Services:
  GET    /api/v1/services               -> list[ServiceRead]
  POST   /api/v1/services               -> ServiceRead (201)
  PUT    /api/v1/services/{id}           -> ServiceRead
  DELETE /api/v1/services/{id}           -> 204

Schedule:
  GET    /api/v1/schedule                -> list[ScheduleDayRead]
  PUT    /api/v1/schedule                -> list[ScheduleDayRead]  (body: ScheduleTemplate)
  GET    /api/v1/schedule/exceptions     -> list[ScheduleExceptionRead]
  POST   /api/v1/schedule/exceptions     -> ScheduleExceptionRead (201)
  DELETE /api/v1/schedule/exceptions/{id} -> 204

Bookings:
  GET    /api/v1/bookings                -> BookingListResponse (query: date_from, date_to, status)
  POST   /api/v1/bookings/manual         -> BookingRead (201)
  PUT    /api/v1/bookings/{id}/cancel    -> BookingRead
  PUT    /api/v1/bookings/{id}/reschedule -> BookingRead

Clients:
  GET    /api/v1/clients                 -> list[MasterClientRead]
  GET    /api/v1/clients/{id}            -> ClientDetailRead

Settings:
  GET    /api/v1/settings                -> MasterSettings
  PUT    /api/v1/settings                -> MasterSettings

Types:
  ServiceRead: { id, name, description, duration_minutes, price, category, is_active, sort_order }
  ScheduleDayRead: { day_of_week, start_time, end_time, break_start, break_end, is_working }
  ScheduleExceptionRead: { id, exception_date, is_day_off, start_time, end_time, reason }
  BookingRead: { id, starts_at, ends_at, status, service_name, client_name, client_phone }
  MasterClientRead: { client: { id, phone, name }, first_visit_at, last_visit_at, visit_count }
  ClientDetailRead: { client: {...}, bookings: [...], visit_count }
</interfaces>
