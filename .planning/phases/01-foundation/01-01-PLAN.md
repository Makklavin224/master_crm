async def lifespan(app: FastAPI):

    await engine.dispose()

**1. `backend/app/services/phone_service.py`** -- E.164 normalization (exact pattern from RESEARCH.md Pattern 5):
```python
import phonenumbers
from phonenumbers import NumberParseException

def normalize_phone(raw_phone: str, default_region: str = "RU") -> str | None:
    """Normalize a phone number to E.164 format. Returns None if invalid."""
    if not raw_phone or not raw_phone.strip():
        return None
    try:
        parsed = phonenumbers.parse(raw_phone.strip(), default_region)
        if not phonenumbers.is_valid_number(parsed):
            return None
        return phonenumbers.format_number(parsed, phonenumbers.PhoneNumberFormat.E164)
    except NumberParseException:
        return None
```
