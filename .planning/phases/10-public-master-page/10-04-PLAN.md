    2. Update `public/src/pages/MasterPage.tsx` — add a useEffect that sets document.title and meta tags when profile loads:

       Create a helper function `updateMetaTags(profile: MasterProfile)`:
       ```typescript
       function updateMetaTags(profile: MasterProfile) {
         // Title
         const title = profile.specialization
           ? `${profile.name} — ${profile.specialization} | МоиОкошки`
           : `${profile.name} | МоиОкошки`;
         document.title = title;

         // Description
         const parts = [profile.specialization, profile.city].filter(Boolean);
         const ratingText = profile.avg_rating
           ? `★ ${profile.avg_rating} (${profile.review_count} отзывов)`
           : "";
         const description = `${profile.name}${parts.length ? " — " + parts.join(", ") : ""}. ${ratingText} Запишитесь онлайн.`.trim();

         // Set or create meta tags
         setMetaTag("description", description);
         setMetaTag("og:title", title, "property");
         setMetaTag("og:description", description, "property");
         setMetaTag("og:type", "profile", "property");
         setMetaTag("og:site_name", "МоиОкошки", "property");

         // Set og:url to canonical URL
         const url = `https://moiokoshki.ru/m/${profile.username || profile.id}`;
         setMetaTag("og:url", url, "property");

         // Set og:image if avatar exists
         if (profile.avatar_path) {
           const imageUrl = `https://moiokoshki.ru/api/v1/media/${profile.avatar_path}`;
           setMetaTag("og:image", imageUrl, "property");
         }
       }

       function setMetaTag(name: string, content: string, attr: "name" | "property" = "name") {
         let el = document.querySelector(`meta[${attr}="${name}"]`) as HTMLMetaElement | null;
         if (!el) {
           el = document.createElement("meta");
           el.setAttribute(attr, name);
           document.head.appendChild(el);
         }
         el.content = content;
       }
       ```

       Add useEffect in MasterPage:
       ```typescript
       useEffect(() => {
         if (profile) {
           updateMetaTags(profile);
         }
         return () => {
           document.title = "МоиОкошки — Запись онлайн";
         };
       }, [profile]);
       ```

       NOTE: Client-side meta tags work for browser tab titles and some social platforms that execute JS. For full SEO with crawlers that don't execute JS, a server-side solution would be needed (deferred to future — the client-side approach is sufficient for social sharing via Telegram, WhatsApp, VK which all execute JS for link previews).
