       ```typescript
       interface BookingFlowState {
         username: string | null;       // master username from URL
         masterId: string | null;       // master UUID from profile
         step: 1 | 2 | 3 | 4 | 5;
         selectedService: ServiceRead | null;
         selectedDate: string | null;   // "YYYY-MM-DD"
         selectedTime: string | null;   // "HH:MM"
         clientName: string;
         clientPhone: string;
         bookingResult: BookingRead | null;

         setMaster: (username: string, masterId: string) => void;
         selectService: (service: ServiceRead) => void;
         selectDate: (date: string) => void;
         selectTime: (time: string) => void;
         setClientInfo: (name: string, phone: string) => void;
         setBookingResult: (booking: BookingRead) => void;
         reset: () => void;
       }
       ```
       - setMaster: stores both username and masterId
       - selectService: sets service + advances step to 2
       - selectDate: sets date, clears selectedTime, advances step to 3
       - selectTime: sets time (does NOT auto-advance, user clicks "Далее")
       - reset: returns to initial state
