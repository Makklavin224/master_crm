```typescript
// Uses useQuery with queryKey array and queryFn calling apiRequest
export function usePayments(filters) {
  return useQuery<PaymentListResponse>({
    queryKey: ["payments", filters],
    queryFn: () => apiRequest<PaymentListResponse>(`/payments/history?${qs}`),
    staleTime: 30_000,
  });
}
```

    **Step 2: Create `web/src/api/analytics.ts`**

    TypeScript interfaces matching backend schemas:
    ```
    AnalyticsSummary { revenue, booking_count, client_count, new_clients, repeat_clients, utilization, avg_check, retention_rate, cancel_rate, noshow_rate }
    RevenueChartPoint { date, revenue }
    TopServiceRow { service_name, booking_count, revenue, percentage }
    DailyBreakdownRow { date, booking_count, revenue, utilization }
    ```
    All money fields are `number` (kopecks).

    4 react-query hooks, all accepting `{ date_from: string; date_to: string }`:
    - `useAnalyticsSummary(params)` -> queryKey: `["analytics", "summary", params]`
    - `useRevenueChart(params)` -> queryKey: `["analytics", "revenue-chart", params]`
    - `useTopServices(params)` -> queryKey: `["analytics", "top-services", params]`
    - `useDailyBreakdown(params)` -> queryKey: `["analytics", "daily", params]`
    Each uses `apiRequest<T>(`/analytics/{endpoint}?date_from=${params.date_from}&date_to=${params.date_to}`)`.
    staleTime: 60_000 (1 minute -- analytics data doesn't change rapidly).

    CSV export function `exportAnalyticsCsv(data: DailyBreakdownRow[], topServices: TopServiceRow[])`:
    - Build CSV string client-side with BOM prefix ("\uFEFF")
    - Section 1: "Топ услуг" header row, then columns: Услуга, Записей, Доход (руб), % от общего
    - Section 2: blank line, "По дням" header row, then columns: Дата, Записей, Доход (руб), Загруженность %
    - Money values: divide by 100 for display in rubles
    - Download via Blob + URL.createObjectURL + click (same pattern as payments CSV)
